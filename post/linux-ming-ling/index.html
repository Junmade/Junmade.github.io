Linux命令

2019-05-29
1. scp

  • 文件

     -- 上传本地文件到服务器
 scp /path/filename username@servername:/path;
 
 -- 从服务器上下载文件
 scp username@servername:/path/filename /path;

  • 目录

    -- 上传目录到服务器
scp -r /path username@servername:/path
-- 从服务器下载目录
scp -r username@servername:/path /path

2. tar

-- 打包
tar -cvf *.tar filename

-- 解压
tar -xvf *.tar

-- 查看多少位
getconf LONG_BIT
